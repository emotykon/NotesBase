1. Mapowanie: Anonimowe vs shm_open

Podstawowe mechanizmy współdzielenia struktur danych.

1.1 Mapowanie Anonimowe (Dla Fork)

// Najszybszy sposób dla procesów rodzic-dziecko.
void* addr = mmap(NULL, size, PROT_READ | PROT_WRITE, MAP_SHARED | MAP_ANONYMOUS, -1, 0);

1.2 Pamięć Współdzielona POSIX (shm_open)

// Dla niezależnych programów. Widoczne w /dev/shm/.
int fd = shm_open("/moj_shm", O_CREAT | O_RDWR, 0666);
ftruncate(fd, size);
int* data = mmap(NULL, size, PROT_READ | PROT_WRITE, MAP_SHARED, fd, 0);

2. Semafory (Nazwane i Nienazwane)

2.1 Semafor Nienazwany (Wewnątrz mmap)

// Inicjalizacja w strukturze: sem_init(&(shm->bin_sem), 1, 1);
// pshared=1 oznacza dzielenie MIĘDZY PROCESAMI.
sem_wait(&(shm->bin_sem)); // Sekcja krytyczna
shm->value++;
sem_post(&(shm->bin_sem));

2.1 Strażnik: Semafor Nazwany (Najbezpieczniejsza metoda)

Zapewnia, że proces "tworzenie + inicjalizacja" jest operacją niepodzielną. Zapobiega Race Condition na starcie.
#include <fcntl.h>
#include <sys/mman.h>
#include <semaphore.h>
#include <stdio.h>
#include <errno.h>

#define SHM_NAME "/moje_obliczenia"
#define SEM_NAME "/moj_straznik"

// 1. Otwórz lub stwórz semafor (wartość początkowa 1)
sem_t *sem = sem_open(SEM_NAME, O_CREAT, 0666, 1);

// 2. Czekaj na dostęp (blokada dla innych procesów)
sem_wait(sem);

// --- SEKCJA KRYTYCZNA START ---
int fd = shm_open(SHM_NAME, O_CREAT | O_EXCL | O_RDWR, 0666);
int first_one = 0;

if (fd >= 0) {
    first_one = 1;
    ftruncate(fd, sizeof(struct shared_data));
    printf("Jestem procesem nr 1, inicjalizuję pamięć...\n");
} else if (errno == EEXIST) {
    fd = shm_open(SHM_NAME, O_RDWR, 0666);
    printf("Dołączam do istniejącej pamięci...\n");
}

void *addr = mmap(NULL, sizeof(struct shared_data), PROT_READ | PROT_WRITE, MAP_SHARED, fd, 0);

if (first_one) {
    // TUTAJ inicjalizujesz muteksy i zerujesz liczniki
}
// --- SEKCJA KRYTYCZNA KONIEC ---

// 3. Puść semafor, by inne procesy mogły dołączyć
sem_post(sem);

3. Bariery Współdzielone

Synchronizacja etapowa. Uwaga: Brak trybu Robust! Jeden padnięty proces = Deadlock wszystkich.
pthread_barrierattr_t attr;
pthread_barrierattr_init(&attr);
pthread_barrierattr_setpshared(&attr, PTHREAD_PROCESS_SHARED);
pthread_barrier_init(&data->barrier, &attr, num_processes);

3.1 Szybki przykład: Bariera dla Wątków

Najprostsza synchronizacja wątków w obrębie jednego procesu.
pthread_barrier_t barrier;

// 1. Inicjalizacja: Czekamy na dokładnie 3 wątki
pthread_barrier_init(&barrier, NULL, 3);

// 2. Wewnątrz funkcji wątku:
printf("Wątek %d dotarł do bariery.\n", id);

// Punkt synchronizacji - tu wątek stoi, dopóki nie będzie kompletu
pthread_barrier_wait(&barrier); 

printf("Wątek %d rusza dalej, bo barierę przekroczyły już 3 wątki!\n", id);

// 3. Na końcu (w main):
pthread_barrier_destroy(&barrier);

4. Mutexy i Robustness (Odporność na śmierć)

Kluczowe przy używaniu abort() lub gdy proces ginie trzymając blokadę.
pthread_mutexattr_setrobust(&attr, PTHREAD_MUTEX_ROBUST);

int rc = pthread_mutex_lock(&shm->mutex);
if (rc == EOWNERDEAD) {
    shm->process_count--; // Korekta danych
    pthread_mutex_consistent(&shm->mutex); // Uzdrowienie
}

5. Sygnały i "Kulturalne" Wyjście

Używamy volatile sig_atomic_t, aby bezpiecznie kończyć pętle.
volatile sig_atomic_t keep_running = 1;
void handle_sigint(int sig) { keep_running = 0; }

int main() {
    signal(SIGINT, handle_sigint);
    while(keep_running) { /* praca */ }
    // Tu sprzątanie
}

6. Workflow: Inicjalizacja i Ostatni Proces

Zasada: Pierwszy inicjuje, ostatni gasi światło (unlink).
  • Otwórz semafor nazwany i zrób sem_wait.
  • Otwórz SHM z O_EXCL aby wykryć lidera.
  • Zrób sem_post. Zwiększ process_count++ w SHM.
  • Na końcu process_count--. Jeśli == 0 -> shm_unlink i sem_unlink.

11. Pełne Rozwiązanie: Agregacja (Zliczanie znaków)

Wydajne zliczanie znaków w dużym pliku przy użyciu mmap oraz fork. Agregacja w SHM za pomocą operacji atomowych __sync_fetch_and_add.
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <fcntl.h>
#include <sys/mman.h>
#include <sys/stat.h>
#include <sys/wait.h>
#include <unistd.h>
#include <time.h>

#define SHM_NAME "/char_count_shm"
#define NUM_CHARS 256

typedef struct {
    long counts[NUM_CHARS];
} shared_data_t;

int main(int argc, char *argv[]) {
    if (argc < 3) { fprintf(stderr, "Użycie: %s <plik> <liczba_procesów>\n", argv[0]); exit(1); }

    const char *file_path = argv[1];
    int N = atoi(argv[2]);
    if (N <= 0) N = 1;

    int fd = open(file_path, O_RDONLY);
    struct stat st;
    fstat(fd, &st);
    size_t file_size = st.st_size;

    int shm_fd = shm_open(SHM_NAME, O_CREAT | O_RDWR | O_TRUNC, 0666);
    ftruncate(shm_fd, sizeof(shared_data_t));
    shared_data_t *shm_ptr = mmap(NULL, sizeof(shared_data_t), PROT_READ | PROT_WRITE, MAP_SHARED, shm_fd, 0);
    memset(shm_ptr->counts, 0, sizeof(shm_ptr->counts));

    size_t chunk_size = file_size / N;
    for (int i = 0; i < N; i++) {
        if (fork() == 0) {
            char *file_mem = mmap(NULL, file_size, PROT_READ, MAP_PRIVATE, fd, 0);
            size_t start = i * chunk_size;
            size_t end = (i == N - 1) ? file_size : (i + 1) * chunk_size;

            long local_counts[NUM_CHARS] = {0};
            for (size_t j = start; j < end; j++) {
                local_counts[(unsigned char)file_mem[j]]++;
            }

            // Agregacja atomowa (Etap 3)
            for (int k = 0; k < NUM_CHARS; k++) {
                __sync_fetch_and_add(&shm_ptr->counts[k], local_counts[k]);
            }
            exit(0);
        }
    }

    // Obsługa błędów (Etap 4)
    int status, success = 1;
    for (int i = 0; i < N; i++) {
        wait(&status);
        if (!WIFEXITED(status) || WEXITSTATUS(status) != 0) success = 0;
    }

    if (success) {
        for (int i = 0; i < NUM_CHARS; i++) {
            if (shm_ptr->counts[i] > 0) printf("Znak '%c': %ld\n", i, shm_ptr->counts[i]);
        }
    }
    
    shm_unlink(SHM_NAME);
    return 0;
}
