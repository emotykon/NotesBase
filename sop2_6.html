1. Semafory Nazwane (sem_open)

Najprostszy przykład użycia semaforów nazwanych. To idealne rozwiązanie, gdy masz dwa różne programy, które muszą zsynchronizować dostęp do czegoś wspólnego. Semafor nazwany żyje w systemie operacyjnym jako "plik" w pamięci.

Program 1: Użytkownik zasobu

#include <stdio.h>
#include <fcntl.h>
#include <semaphore.h>
#include <unistd.h>

int main() {
    // 1. Otwieramy/Tworzymy semafor. Nazwa musi zaczynać się od '/'
    // Wartość 1 oznacza: tylko jeden proces może przejść naraz.
    sem_t *sem = sem_open("/szlaban", O_CREAT, 0666, 1);
    if (sem == SEM_FAILED) { perror("sem_open"); return 1; }

    printf("Stoję przed szlabanem...\n");
    
    // 2. Czekanie (P) - jeśli wartość wynosi 1, zmienia na 0 i wchodzi.
    sem_wait(sem);

    printf("Wjechałem! Robię coś ważnego przez 5 sekund...\n");
    sleep(5);

    printf("Wyjeżdżam i podnoszę szlaban.\n");
    
    // 3. Sygnalizacja (V) - zmienia wartość z 0 na 1, budząc następnego.
    sem_post(sem);

    // Zamykamy dostęp w tym procesie
    sem_close(sem);
    return 0;
}

Program 2: Sprzątacz

#include <semaphore.h>

int main() {
    // Usuwa semafor z systemu operacyjnego
    sem_unlink("/szlaban");
    return 0;
}

2. Mapowanie Plików i Pamięci Dzielonej

Oto zestaw dwóch "szablonowych" funkcji. Pierwsza pod pliki (Bitronka), druga pod RAM (Całkowanie). Najważniejsze to zapamiętać: ftruncate jest obowiązkowe!

2.1 Mapowanie pliku na dysku (Bitronka)

#include <fcntl.h>
#include <sys/mman.h>
#include <unistd.h>
#include <stdio.h>

void* map_disk_file(const char* filename, size_t size) {
    int fd = open(filename, O_CREAT | O_RDWR | O_TRUNC, 0666);
    if (fd == -1) return NULL;

    if (ftruncate(fd, size) == -1) {
        close(fd);
        return NULL;
    }

    void* ptr = mmap(NULL, size, PROT_READ | PROT_WRITE, MAP_SHARED, fd, 0);
    close(fd);

    if (ptr == MAP_FAILED) return NULL;
    return ptr;
}

void* setup_named_shm(const char* shm_name, size_t size) {

2.3 Anonimowa Pamięć Współdzielona (Prosty przykład fork)

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/mman.h>
#include <sys/wait.h>
#include <pthread.h>

struct shared_data {
    pthread_mutex_t mut;
    int results[2];
};

int main() {
    // Mapujemy całą strukturę naraz anonimowo (bez pliku/nazwy)
    struct shared_data* data = mmap(NULL, sizeof(struct shared_data), 
                                    PROT_READ | PROT_WRITE, 
                                    MAP_SHARED | MAP_ANONYMOUS, -1, 0);

    pthread_mutexattr_t attr;
    pthread_mutexattr_init(&attr);
    pthread_mutexattr_setpshared(&attr, PTHREAD_PROCESS_SHARED);
    
    pthread_mutex_init(&(data->mut), &attr);
    data->results[0] = 0;
    data->results[1] = 0;

    for (int i = 0; i < 5; i++) {
        if (fork() == 0) {
            for (int j = 0; j < 100000; j++) {
                pthread_mutex_lock(&(data->mut));
                data->results[1]++;
                pthread_mutex_unlock(&(data->mut));
            }
            exit(0);
        }
    }

    for (int i = 0; i < 5; i++) wait(NULL);

    printf("Wynik: %d\n", data->results[1]);

    pthread_mutex_destroy(&(data->mut));
    pthread_mutexattr_destroy(&attr);
    munmap(data, sizeof(struct shared_data));
    return 0;
}

3. Inicjalizacja Tablicy Mutexów

Niezbędna, gdy pamięć zawiera tablicę kłódek (np. po jednej dla każdej półki w Bitronce).
void init_all_mutexes(pthread_mutex_t* array, int count) {
    pthread_mutexattr_t attr;
    pthread_mutexattr_init(&attr);
    pthread_mutexattr_setpshared(&attr, PTHREAD_PROCESS_SHARED);
    pthread_mutexattr_setrobust(&attr, PTHREAD_MUTEX_ROBUST);

    for (int i = 0; i < count; i++) {
        if (pthread_mutex_init(&array[i], &attr) != 0) {
            perror("pthread_mutex_init failed");
        }
    }
    pthread_mutexattr_destroy(&attr);
}

// Bezpieczny swap z unikaniem deadlocka
void worker_swap(pthread_mutex_t* mutexes, int idx1, int idx2) {
    int first = (idx1 < idx2) ? idx1 : idx2;
    int second = (idx1 < idx2) ? idx2 : idx1;

    if (pthread_mutex_lock(&mutexes[first]) == EOWNERDEAD) {
        pthread_mutex_consistent(&mutexes[first]);
    }
    if (pthread_mutex_lock(&mutexes[second]) == EOWNERDEAD) {
        pthread_mutex_consistent(&mutexes[second]);
    }

    // --- SWAP ---

    pthread_mutex_unlock(&mutexes[second]);
    pthread_mutex_unlock(&mutexes[first]);
}

4. Etap 2: Procesy i Logika Pracy

Najważniejsza rzecz tutaj to "Fork Safety" – upewnienie się, że dzieci kończą się przez exit(0).
void spawn_workers(int num_workers, shared_t* sh) {
    for (int i = 0; i < num_workers; i++) {
        pid_t pid = fork();
        if (pid < 0) { perror("fork"); exit(1); }

        if (pid == 0) {
            printf("[PID %d] Worker %d startuje...\n", getpid(), i);
            worker_main_loop(i, sh);
            exit(0); // Krytyczne!
        }
    }
}

void worker_main_loop(int id, shared_t* sh) {
    for (int i = 0; i < 10 && keep_running && !sh->end_flag; i++) {
        
        // Sekcja Krytyczna
        safe_lock(&sh->mutex); 
        sh->total_counter++;
        printf("[Worker %d] Paczka %d wysłana.\n", id, i);
        pthread_mutex_unlock(&sh->mutex);
        
        ms_sleep(100); 
    }
    printf("[PID %d] Worker kończy zmianę.\n", getpid());
}

}

6. Etap 4: Robustness i Odporność na Błędy

Obsługa awarii procesu trzymającego mutex (EOWNERDEAD) oraz ostateczne sprzątanie.
void safe_lock(pthread_mutex_t* m, shared_t* sh) {
    int rc = pthread_mutex_lock(m);
    if (rc == EOWNERDEAD) {
        printf("[PID %d] Wykryto śmierć właściciela mutexa!\n", getpid());
        pthread_mutex_consistent(m);
        if (sh != NULL) {
            sh->alive_workers--;
        }
    } else if (rc != 0) {
        perror("pthread_mutex_lock");
        exit(1);
    }
}

// Zwalnianie zasobów (Całkowanie L3)
void handle_exit_logic(shared_t* sh) {
    safe_lock(&sh->mutex, sh);
    sh->processes_count--;

    if (sh->processes_count == 0) {
        double result = (double)sh->hit_points / sh->total_points * (ZAKRES_CALKI);

        shm_unlink(SHM_NAME);
        sem_unlink(SEM_NAME);
    } else {
        pthread_mutex_unlock(&sh->mutex);

    }
}

7. Kompendium Flag i Atrybutów

Flaga Zastosowanie Kiedy użyć?
O_CREAT Tworzy obiekt, jeśli nie istnieje. Zawsze przy inicjalizacji zasobów.
O_RDWR Otwiera w trybie Odczyt + Zapis. Zawsze dla pamięci współdzielonej (SHM).
O_EXCL Zwraca błąd, jeśli plik już istnieje. W Etapie 1, aby wykryć "Pierwszy Proces".
O_TRUNC Czyści zawartość pliku do zera. W Bitronce przy otwieraniu pliku sklepu.
  • MAP_SHARED: Zmiany w pamięci są widoczne dla innych procesów.
  • PTHREAD_PROCESS_SHARED: Mutex działa w pamięci dzielonej między procesami.
  • PTHREAD_MUTEX_ROBUST: Zwraca EOWNERDEAD kolejnemu chętnemu po nagłej śmierci.

#define SHOP_FILENAME "./shop"
#define MIN_SHELVES 8
#define MAX_SHELVES 256
#define MIN_WORKERS 1
#define MAX_WORKERS 64

#define ERR(source)                                     \
    do                                                  \
    {                                                   \
        fprintf(stderr, "%s:%d\n", __FILE__, __LINE__); \
        perror(source);                                 \
        kill(0, SIGKILL);                               \
        exit(EXIT_FAILURE);                             \
    } while (0)

#define SWAP(x, y)         \
    do                     \
    {                      \
        typeof(x) __x = x; \
        typeof(y) __y = y; \
        x = __y;           \
        y = __x;           \
    } while (0)

typedef struct shop
{
    int* shelves;
    pthread_mutex_t* shelf_mutexes;
    int* alive_workers;
    pthread_mutex_t* alive_workers_mutex;
    int* do_work;
    pthread_mutex_t* do_work_mutex;
    size_t mem_size;
    int shelves_count;
} shop_t;

void usage(char* program_name)
{
    fprintf(stderr, "Usage: \n");
    fprintf(stderr, "\t%s n m\n", program_name);
    exit(EXIT_FAILURE);
}

void shuffle(int* array, int n)
{
    for (int i = n - 1; i > 0; i--)
    {
        int j = rand() % (i + 1);
        SWAP(array[i], array[j]);
    }
}

void ms_sleep(unsigned int milli)
{
    time_t sec = (int)(milli / 1000);
    milli = milli - (sec * 1000);
    struct timespec ts = {0};
    ts.tv_sec = sec;
    ts.tv_nsec = milli * 1000000L;
    if (nanosleep(&ts, &ts))
        ERR("nanosleep");
}

shop_t open_shop(int shelves_count, int workers_count)
{
    size_t shop_size = shelves_count * sizeof(int);
    shop_t shop = {.shelves_count = shelves_count};

    int shop_fd = open(SHOP_FILENAME, O_CREAT | O_RDWR | O_TRUNC, 0600);
    if (shop_fd == -1) ERR("open");
    if (ftruncate(shop_fd, (off_t)shop_size)) ERR("ftruncate");

    shop.shelves = mmap(NULL, shop_size, PROT_WRITE | PROT_READ, MAP_SHARED, shop_fd, 0);
    if (shop.shelves == MAP_FAILED) ERR("mmap");
    if (close(shop_fd)) ERR("close");

    shop.mem_size = (shelves_count + 2) * sizeof(pthread_mutex_t) + 2 * sizeof(int);
    if ((shop.shelf_mutexes = mmap(NULL, shop.mem_size, PROT_READ | PROT_WRITE, MAP_SHARED | MAP_ANONYMOUS, -1, 0)) == MAP_FAILED)
        ERR("mmap");

    shop.do_work_mutex = shop.shelf_mutexes + shelves_count;
    shop.alive_workers_mutex = shop.do_work_mutex + 1;
    shop.do_work = (int*)(shop.alive_workers_mutex + 1);
    *shop.do_work = 1;
    shop.alive_workers = shop.do_work + 1;
    *shop.alive_workers = workers_count * 2;

    pthread_mutexattr_t mutex_attr;
    pthread_mutexattr_init(&mutex_attr);
    pthread_mutexattr_setpshared(&mutex_attr, PTHREAD_PROCESS_SHARED);
    pthread_mutexattr_setrobust(&mutex_attr, PTHREAD_MUTEX_ROBUST);

    for (int i = 0; i < shelves_count + 2; ++i)
    {
        shop.shelves[i] = i + 1;
        pthread_mutex_init(shop.shelf_mutexes + i, &mutex_attr);
    }

    shuffle(shop.shelves, shop.shelves_count);
    pthread_mutexattr_destroy(&mutex_attr);

    return shop;
}

void close_shop(shop_t* shop, int main_thread)
{
    if (main_thread)
    {
        for (int i = 0; i < shop->shelves_count + 2; ++i)
        {
            pthread_mutex_destroy(shop->shelf_mutexes + i);
        }
        msync(shop->shelves, shop->shelves_count * sizeof(int), MS_SYNC);
    }
    munmap(shop->shelves, shop->shelves_count * sizeof(int));
    munmap(shop->shelf_mutexes, shop->mem_size);
}

void print_array(int* array, int n)
{
    for (int i = 0; i < n; ++i) printf("%3d ", array[i]);
    printf("\n");
}

void lock_shelf_mutex(shop_t* shop, int i)
{
    switch (pthread_mutex_lock(shop->shelf_mutexes + i))
    {
        case 0:
            break;
        case EOWNERDEAD:
            printf("[%d] Found a dead body in aisle %d\n", getpid(), i + 1);
            pthread_mutex_consistent(shop->shelf_mutexes + i);
            pthread_mutex_lock(shop->alive_workers_mutex);
            (*shop->alive_workers)--;
            pthread_mutex_unlock(shop->alive_workers_mutex);
            break;
        default:
            ERR("pthread_mutex_lock");
    }
}

void worker_routine(shop_t* shop)
{
    srand(getpid());
    printf("[%d] Worker reports for a night shift\n", getpid());

    while (1)
    {
        pthread_mutex_lock(shop->do_work_mutex);
        if (!(*shop->do_work))
        {
            pthread_mutex_unlock(shop->do_work_mutex);
            break;
        }
        pthread_mutex_unlock(shop->do_work_mutex);

        int i = rand() % shop->shelves_count;
        int j = rand() % shop->shelves_count;
        while (i == j) j = rand() % shop->shelves_count;

        if (i > j) SWAP(i, j);

        lock_shelf_mutex(shop, i);
        lock_shelf_mutex(shop, j);

        ms_sleep(100);

        if (rand() % 100 == 0)
        {
            printf("[%d] Trips over pallet and dies\n", getpid());
            abort();
        }

        if (shop->shelves[i] > shop->shelves[j])
        {
            SWAP(shop->shelves[i], shop->shelves[j]);
        }

        pthread_mutex_unlock(shop->shelf_mutexes + j);
        pthread_mutex_unlock(shop->shelf_mutexes + i);
    }

    close_shop(shop, 0);
    exit(EXIT_SUCCESS);
}

void manager_routine(shop_t* shop)
{
    printf("[%d] Manager reports for a night shift\n", getpid());

    while (1)
    {
        ms_sleep(500);

        for (int i = 0; i < shop->shelves_count; ++i) lock_shelf_mutex(shop, i);

        print_array(shop->shelves, shop->shelves_count);

        int sorted = 1;
        for (int i = 0; i < shop->shelves_count; ++i)
        {
            if (i + 1 != shop->shelves[i])
            {
                sorted = 0;
                break;
            }
        }

        msync(shop->shelves, shop->shelves_count * sizeof(int), MS_SYNC);
        for (int i = 0; i < shop->shelves_count; ++i)
        {
            pthread_mutex_unlock(shop->shelf_mutexes + i);
        }

        if (sorted)
        {
            printf("[%d] The shop shelves are sorted\n", getpid());
            break;
        }

        pthread_mutex_lock(shop->alive_workers_mutex);
        if (*shop->alive_workers == 0)
        {
            printf("[%d] All workers died, I hate my job\n", getpid());
            break;
        }
        else
        {
            printf("[%d] Workers alive: %d\n", getpid(), *shop->alive_workers / 2);
        }
        pthread_mutex_unlock(shop->alive_workers_mutex);
    }

    pthread_mutex_lock(shop->do_work_mutex);
    *shop->do_work = 0;
    pthread_mutex_unlock(shop->do_work_mutex);
    close_shop(shop, 0);
    exit(EXIT_SUCCESS);
}

void create_workers(int workers_count, shop_t* shop)
{
    pid_t pid;
    for (int i = 0; i < workers_count; ++i)
    {
        pid = fork();
        if (pid == 0) worker_routine(shop);
    }
    pid = fork();
    if (pid == 0) manager_routine(shop);
}

int main(int argc, char** argv)
{
    if (argc != 3) usage(argv[0]);

    int shelves_count = atoi(argv[1]);
    int workers_count = atoi(argv[2]);

    shop_t shop = open_shop(shelves_count, workers_count);

    create_workers(workers_count, &shop);

    while (wait(NULL) > 0);

    close_shop(&shop, 1);
    printf("Night shift in Bitronka is over\n");

    return EXIT_SUCCESS;
}

#define SHMEM_SEMAPHORE_NAME "/sop-shmem-sem"
#define SHMEM_NAME "/sop-shmem"
#define UNUSED(x) (void)(x)
#define ERR(source) do { perror(source); exit(EXIT_FAILURE); } while(0)

typedef struct shared_result
{
    pthread_mutex_t process_counter_lock;
    int process_counter;
    pthread_mutex_t hit_lock;
    uint64_t hit_count;
    uint64_t total_count;
    double segment_start;
    double segment_end;
} shared_result_t;

shared_result_t* shared_results_init(double seg_start, double seg_end)
{
    sem_t* shmem_init_sem = sem_open(SHMEM_SEMAPHORE_NAME, O_CREAT, S_IRUSR | S_IWUSR, 1);
    if (shmem_init_sem == SEM_FAILED) ERR("sem_open");

    struct timespec ts;
    if (clock_gettime(CLOCK_REALTIME, &ts) == -1) ERR("clock_gettime");
    ts.tv_sec += 10;
    
    int s = 0;
    while ((s = sem_timedwait(shmem_init_sem, &ts)) == -1 && errno == EINTR) continue;

    if (s == -1) ERR("sem_timedwait");

    int initialize = 1;
    int shmem_fd = shm_open(SHMEM_NAME, O_CREAT | O_EXCL | O_RDWR, S_IRUSR | S_IWUSR);
    if (shmem_fd == -1)
    {
        if (errno != EEXIST) { sem_post(shmem_init_sem); ERR("shm_open"); }
        shmem_fd = shm_open(SHMEM_NAME, O_RDWR, S_IRUSR | S_IWUSR);
        if (shmem_fd == -1) { sem_post(shmem_init_sem); ERR("shm_open"); }
        initialize = 0;
    }

    if (initialize && ftruncate(shmem_fd, sizeof(shared_result_t)) == -1)
    {
        shm_unlink(SHMEM_NAME);
        sem_post(shmem_init_sem);
        ERR("ftruncate");
    }

    shared_result_t* shmem = mmap(NULL, sizeof(shared_result_t), PROT_READ | PROT_WRITE, MAP_SHARED, shmem_fd, 0);
    close(shmem_fd);

    if (initialize)
    {
        pthread_mutexattr_t attr2;
        pthread_mutexattr_init(&attr2);
        pthread_mutexattr_setpshared(&attr2, PTHREAD_PROCESS_SHARED);
        pthread_mutex_init(&shmem->process_counter_lock, &attr2);
        pthread_mutex_init(&shmem->hit_lock, &attr2);

        shmem->total_count = 0;
        shmem->hit_count = 0;
        shmem->process_counter = 1;
        shmem->segment_start = seg_start;
        shmem->segment_end = seg_end;
    }
    else
    {
        pthread_mutex_lock(&shmem->process_counter_lock);
        shmem->process_counter++;
        pthread_mutex_unlock(&shmem->process_counter_lock);
    }
    sem_post(shmem_init_sem);
    return shmem;
}

void shared_results_close(shared_result_t* shmem) { munmap(shmem, sizeof(shared_result_t)); }

void shared_results_destroy(shared_result_t* shmem)
{
    pthread_mutex_destroy(&shmem->process_counter_lock);
    pthread_mutex_destroy(&shmem->hit_lock);
    shared_results_close(shmem);
    shm_unlink(SHMEM_NAME);
    sem_unlink(SHMEM_SEMAPHORE_NAME);
}

void sigint_handler(int signo) { UNUSED(signo); working = 0; }

typedef struct calculation_result
{
    uint64_t hit_count;
    uint64_t point_count;
} calculation_result_t;

double func(double x)
{
    usleep(1000);
    return exp(-x * x);
}

void randomize_points(calculation_result_t* res, shared_result_t* params)
{
    uint64_t i = 0;
    for (; i < res->point_count && working; ++i)
    {
        double rand_x = ((double)rand() / RAND_MAX) * (params->segment_end - params->segment_start) + params->segment_start;
        double rand_y = ((double)rand() / RAND_MAX);
        double real_y = func(rand_x);
        if (rand_y <= real_y) res->hit_count++;
    }
    if (!working) res->point_count = i;
}

double summarize_calculations(shared_result_t* params)
{
    return (params->segment_end - params->segment_start) * ((double)params->hit_count / (double)params->total_count);
}

void usage(char* argv[])
{
    printf("%s [options]\n", argv[0]);
}

int main(int argc, char* argv[])
{
    double seg_start = -1;
    double seg_end = 1;
    uint64_t batch_size = 1000;

    int c;
    while ((c = getopt(argc, argv, "s:e:b:")) != -1)
    {
        switch (c)
        {
            case 's': seg_start = (double)strtof(optarg, NULL); break;
            case 'e': seg_end = (double)strtof(optarg, NULL); break;
            case 'b': batch_size = (uint64_t)atoll(optarg); break;
            default: usage(argv); exit(EXIT_FAILURE);
        }
    }

    sethandler(sigint_handler, SIGINT);
    srand(time(NULL) * getpid());

    shared_result_t* shmem = shared_results_init(seg_start, seg_end);

    while (working)
    {
        printf("Taking %lu points for calcualtion\n", batch_size);
        calculation_result_t res;
        res.point_count = batch_size;
        res.hit_count = 0;

        randomize_points(&res, shmem);

        pthread_mutex_lock(&shmem->hit_lock);
        printf("Adding %lu probes. Hit %lu points.\n", res.point_count, res.hit_count);
        shmem->hit_count += res.hit_count;
        shmem->total_count += res.point_count;
        pthread_mutex_unlock(&shmem->hit_lock);
    }

    pthread_mutex_lock(&shmem->process_counter_lock);
    int counter_res = --shmem->process_counter;
    pthread_mutex_unlock(&shmem->process_counter_lock);

    if (!counter_res)
    {
        double res = summarize_calculations(shmem);
        printf("Integral approx = %f\n", res);
        shared_results_destroy(shmem);
    }
    else
    {
        shared_results_close(shmem);
    }

    return EXIT_SUCCESS;
}
